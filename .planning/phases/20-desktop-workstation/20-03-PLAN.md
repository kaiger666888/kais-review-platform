Expanded `workstationState()` function:
```javascript
function workstationState() {
  return {
    activeShotId: null,
    activeShot: null,
    allShots: [],          // populated from shot queue DOM
    allShotIds: [],        // ordered list of shot IDs
    selectedItems: [],
    lastClickedIndex: null,
    showComparison: false,
    comparisonMode: 'first-last',  // 'first-last' | 'current-history' | 'current-reference'
    showPolicy: false,
    showLog: false,
    batchMode: false,
    leftCollapsed: false,
    rightCollapsed: false,

    // Keyboard input guard — skip shortcuts when typing in inputs
    isInputFocused() {
      const tag = document.activeElement?.tagName?.toLowerCase();
      return tag === 'input' || tag === 'textarea' || tag === 'select';
    },

    handleKey(event, key) {
      if (this.isInputFocused()) return;

      switch(key) {
        case 'space': this.togglePlay(); break;
        case 'y': this.approve(); break;
        case 'n': this.reject(); break;
        case 'j': this.nextShot(); break;
        case 'k': this.prevShot(); break;
        case 'd': this.toggleComparison(); break;
        case 'b': this.toggleBatchMode(); break;
        case 'g': this.togglePolicy(); break;
        case 'escape': this.resetMode(); break;
      }
    },

    // Navigation
    nextShot() {
      if (this.allShotIds.length === 0) return;
      const idx = this.allShotIds.indexOf(this.activeShotId);
      const next = idx < this.allShotIds.length - 1 ? idx + 1 : 0;
      this.selectShotById(this.allShotIds[next]);
    },

    prevShot() {
      if (this.allShotIds.length === 0) return;
      const idx = this.allShotIds.indexOf(this.activeShotId);
      const prev = idx > 0 ? idx - 1 : this.allShotIds.length - 1;
      this.selectShotById(this.allShotIds[prev]);
    },

    selectShotById(id) {
      this.activeShotId = id;
      // Trigger HTMX to load the shot detail
      htmx.ajax('GET', `/partials/shot-card-detail/${id}`, '#decision-panel');
    },

    // Media
    togglePlay() {
      const video = document.getElementById('shot-video');
      if (!video) return;
      if (video.paused) video.play(); else video.pause();
    },

    // Decisions
    approve() {
      if (!this.activeShotId) return;
      if (this.batchMode && this.selectedItems.length > 0) {
        this.batchApprove();
      } else {
        htmx.ajax('POST', `/shot-cards/${this.activeShotId}/approve`, '#shot-queue-list');
      }
    },

    reject() {
      if (!this.activeShotId) return;
      if (this.batchMode && this.selectedItems.length > 0) {
        this.batchReject();
      } else {
        // Show reject dialog
        const panel = document.getElementById('decision-panel');
        htmx.ajax('POST', `/shot-cards/${this.activeShotId}/reject`, '#decision-panel');
      }
    },

    // Batch operations
    toggleBatchMode() {
      this.batchMode = !this.batchMode;
      if (!this.batchMode) this.selectedItems = [];
    },

    toggleSelect(shotId, index, event) {
      if (!this.batchMode) {
        // Normal mode: select shot
        this.activeShotId = shotId;
        return;
      }

      if (event.shiftKey && this.lastClickedIndex !== null) {
        // Range select
        const start = Math.min(this.lastClickedIndex, index);
        const end = Math.max(this.lastClickedIndex, index);
        for (let i = start; i <= end; i++) {
          const id = this.allShotIds[i];
          if (id && !this.selectedItems.includes(id)) {
            this.selectedItems.push(id);
          }
        }
      } else if (event.ctrlKey || event.metaKey) {
        // Toggle individual
        const idx = this.selectedItems.indexOf(shotId);
        if (idx > -1) this.selectedItems.splice(idx, 1);
        else this.selectedItems.push(shotId);
      } else {
        // Single select in batch mode (replace selection)
        this.selectedItems = [shotId];
      }
      this.lastClickedIndex = index;
    },

    batchApprove() {
      if (this.selectedItems.length === 0) return;
      // Use fetch to POST batch approve, then refresh queue
      fetch('/partials/shot-cards/batch/approve', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({shot_card_ids: this.selectedItems})
      }).then(() => {
        htmx.ajax('GET', '/partials/shot-queue', '#shot-queue-list');
        this.selectedItems = [];
        this.batchMode = false;
        window.dispatchEvent(new CustomEvent('show-toast', {detail: {message: `${this.selectedItems.length} shot cards approved`, type: 'success'}}));
      });
    },

    batchReject() {
      if (this.selectedItems.length === 0) return;
      const reason = prompt('Rejection reason:');
      if (!reason) return;
      fetch('/partials/shot-cards/batch/reject', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({shot_card_ids: this.selectedItems, reason: reason})
      }).then(() => {
        htmx.ajax('GET', '/partials/shot-queue', '#shot-queue-list');
        this.selectedItems = [];
        this.batchMode = false;
        window.dispatchEvent(new CustomEvent('show-toast', {detail: {message: `${this.selectedItems.length} shot cards rejected`, type: 'success'}}));
      });
    },

    // Comparison
    toggleComparison() {
      this.showComparison = !this.showComparison;
    },

    // Policy
    togglePolicy() {
      this.showPolicy = !this.showPolicy;
    },

    // Reset all modes
    resetMode() {
      this.showComparison = false;
      this.showPolicy = false;
      this.showLog = false;
      this.batchMode = false;
      this.selectedItems = [];
    },

    // Collapse toggles
    toggleLeft() {
      this.leftCollapsed = !this.leftCollapsed;
    },
    toggleRight() {
      this.rightCollapsed = !this.rightCollapsed;
    },

    // Initialize shot IDs from DOM
    init() {
      this.refreshShotIds();
      // Watch for HTMX afterSettle to refresh shot IDs
      document.body.addEventListener('htmx:afterSettle', () => {
        this.refreshShotIds();
      });
    },

    refreshShotIds() {
      // Read shot IDs from the shot queue cards in the DOM
      const cards = document.querySelectorAll('[data-shot-id]');
      this.allShotIds = Array.from(cards).map(el => parseInt(el.dataset.shotId)).filter(Boolean);
    }
  };
}
```
