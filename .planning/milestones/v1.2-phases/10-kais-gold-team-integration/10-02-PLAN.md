    ```python
    @app.post("/callback/review_result", status_code=202)
    async def review_result(
        payload: ReviewResultPayload,
        request: Request,
    ) -> CallbackResponse:
        """Receive review result from review platform.

        When review platform completes a review (approval or rejection),
        it POSTs the result here. The worker's Guardian polls for this
        status change or receives it via event processing.

        Flow: review platform evaluates -> approves/rejects -> callback delivery
              -> this endpoint -> record event -> worker picks up -> resume/fail task
        """
        if not await _verify_hmac(request, payload.model_dump()):
            raise HTTPException(status_code=401, detail="Invalid signature")

        db = _get_db()

        # Record as callback event for processing by CallbackConsumer
        event_id = f"review-{payload.review_id}-{uuid.uuid4().hex[:8]}"
        existing = await db.get_callback_event(event_id)
        if existing:
            logger.info("Duplicate review result %s ignored", event_id)
            return CallbackResponse(status="duplicate", event_id=event_id)

        await db.record_callback_event(
            event_id=event_id,
            event_type="review_result",
            task_id="",  # Will be resolved by consumer via review_id -> task mapping
            payload=payload.model_dump(),
        )

        disposition = payload.disposition or ""
        logger.info(
            "Review result received: review_id=%d disposition=%s",
            payload.review_id, disposition,
        )
        return CallbackResponse(status="accepted", event_id=event_id)
    ```

    ```python
    """Review platform integration — submit tasks for review and poll for results.

    Provides submit_for_review() and poll_review_status() for Guardian to use
    before dispatching GPU tasks. Handles crash recovery via checkpoint.
    """

    import asyncio
    import json
    import logging
    from pathlib import Path
    from typing import Any

    import httpx

    logger = logging.getLogger(__name__)

    # Review platform configuration
    REVIEW_PLATFORM_URL = "http://192.168.71.140:8090"
    CALLBACK_URL = "http://192.168.71.140:8900/callback/review_result"
    POLL_INTERVAL = 30.0  # seconds between status checks
    MAX_POLL_DURATION = 86400.0  # 24 hours max wait

    # Risk tier mapping (matches gold_team_risk.yaml policy)
    HIGH_RISK_TYPES = {
        "blender_render", "custom_script",
        "face_swap", "face_enhance", "frame_enhance", "lip_sync_ff",
        "colorize", "age_modify", "face_edit", "bg_remove", "face_pipeline",
    }

    def _compute_risk_score(task_type: str) -> float:
        """Compute risk score based on task type."""
        if task_type in HIGH_RISK_TYPES:
            return 0.8
        return 0.2


    async def submit_for_review(
        task: dict[str, Any],
        callback_secret: str = "",
        api_key: str = "",
    ) -> dict[str, Any] | None:
        """Submit a GPU task to the review platform.

        Uses the review platform REST API directly (httpx) rather than
        importing ReviewPlatformClient to avoid cross-repo dependency issues.

        Args:
            task: Task dict with task_id, task_type, params, etc.
            callback_secret: HMAC secret for callback verification.
            api_key: Review platform API key for authentication.

        Returns:
            Response dict with review_id, state, routing on success.
            None on failure (logged, not raised).
        """
        task_id = task.get("task_id", "unknown")
        task_type = task.get("task_type", "unknown")

        # Step 1: Get JWT token
        try:
            async with httpx.AsyncClient(timeout=10.0) as client:
                auth_resp = await client.post(
                    f"{REVIEW_PLATFORM_URL}/api/v1/auth/token",
                    json={"api_key": api_key, "client_id": "kais-gold-team"},
                )
                auth_resp.raise_for_status()
                token = auth_resp.json()["data"]["token"]
        except Exception as e:
            logger.error("Review auth failed for task %s: %s", task_id, e)
            return None

        # Step 2: Submit review
        review_payload = {
            "type": "gpu_task",
            "content_ref": task_id,
            "source_system": "kais-gold-team",
            "priority": "normal",
            "risk_score": _compute_risk_score(task_type),
            "metadata": {
                "task_type": task_type,
                "created_by": task.get("created_by", "telegram"),
                "gpu_required": True,
                "params": task.get("params", {}),
                "tags": task.get("tags", []),
            },
            "callback_url": CALLBACK_URL,
            "callback_secret": callback_secret or None,
        }

        try:
            async with httpx.AsyncClient(timeout=10.0) as client:
                resp = await client.post(
                    f"{REVIEW_PLATFORM_URL}/api/v1/reviews/",
                    json=review_payload,
                    headers={"Authorization": f"Bearer {token}"},
                )
                resp.raise_for_status()
                result = resp.json()
                data = result.get("data", {})
                logger.info(
                    "Review submitted: task=%s review_id=%s routing=%s",
                    task_id, data.get("review_id"), data.get("routing"),
                )
                return data
        except Exception as e:
            logger.error("Review submission failed for task %s: %s", task_id, e)
            return None


    async def poll_review_status(
        review_id: int,
        api_key: str = "",
        poll_interval: float = POLL_INTERVAL,
        max_duration: float = MAX_POLL_DURATION,
    ) -> str:
        """Poll the review platform for review status until complete.

        Args:
            review_id: The review platform review ID.
            api_key: Review platform API key for authentication.
            poll_interval: Seconds between polls.
            max_duration: Maximum total poll duration before giving up.

        Returns:
            Final disposition string: "APPROVED" (AUTO also treated as approved),
            "REJECTED", "BLOCKED", or "TIMEOUT".
        """
        import time
        start = time.monotonic()

        # Get JWT token once (refresh if expired during long poll)
        token = await _get_jwt_token(api_key)
        if not token:
            return "TIMEOUT"

        while (time.monotonic() - start) < max_duration:
            try:
                async with httpx.AsyncClient(timeout=10.0) as client:
                    resp = await client.get(
                        f"{REVIEW_PLATFORM_URL}/api/v1/reviews/{review_id}",
                        headers={"Authorization": f"Bearer {token}"},
                    )
                    if resp.status_code == 401:
                        # Token expired, refresh
                        token = await _get_jwt_token(api_key)
                        if not token:
                            return "TIMEOUT"
                        continue

                    resp.raise_for_status()
                    data = resp.json().get("data", {})
                    state = data.get("state", "")
                    disposition = data.get("disposition", "")

                    if state == "COMPLETE":
                        if disposition in ("AUTO", "APPROVED"):
                            return "APPROVED"
                        elif disposition in ("REJECTED", "BLOCK"):
                            return "REJECTED"
                        return disposition or "APPROVED"

                    # Still in progress (PENDING, POLICY_EVAL, APPROVING)
                    logger.debug(
                        "Review %d still in state %s, polling in %ds",
                        review_id, state, poll_interval,
                    )
            except Exception as e:
                logger.warning("Review poll failed for %d: %s", review_id, e)

            await asyncio.sleep(poll_interval)

        logger.error("Review poll timed out for review_id=%d", review_id)
        return "TIMEOUT"


    async def _get_jwt_token(api_key: str) -> str | None:
        """Get a JWT token from the review platform."""
        try:
            async with httpx.AsyncClient(timeout=10.0) as client:
                resp = await client.post(
                    f"{REVIEW_PLATFORM_URL}/api/v1/auth/token",
                    json={"api_key": api_key, "client_id": "kais-gold-team"},
                )
                resp.raise_for_status()
                return resp.json()["data"]["token"]
        except Exception as e:
            logger.error("JWT token request failed: %s", e)
            return None


    def save_review_checkpoint(task_id: str, review_id: int, workspace: Path) -> None:
        """Save review checkpoint for crash recovery.

        Writes a .review_checkpoint file in the task's running directory.
        On Guardian restart, if checkpoint exists, the task can resume polling.
        """
        checkpoint_file = workspace / ".running" / f"{task_id}.review_checkpoint"
        checkpoint_file.write_text(json.dumps({
            "review_id": review_id,
            "task_id": task_id,
        }))
        logger.info("Review checkpoint saved: task=%s review_id=%d", task_id, review_id)


    def load_review_checkpoint(task_id: str, workspace: Path) -> dict | None:
        """Load review checkpoint if it exists."""
        checkpoint_file = workspace / ".running" / f"{task_id}.review_checkpoint"
        if checkpoint_file.exists():
            try:
                return json.loads(checkpoint_file.read_text())
            except (json.JSONDecodeError, OSError):
                return None
        return None


    def clear_review_checkpoint(task_id: str, workspace: Path) -> None:
        """Remove review checkpoint after review completes."""
        checkpoint_file = workspace / ".running" / f"{task_id}.review_checkpoint"
        checkpoint_file.unlink(missing_ok=True)
    ```

    ```python
    async def _handle_review(self, task: dict[str, Any]) -> str:
        """Submit task for review and wait for result.

        Returns: "APPROVED", "REJECTED", "TIMEOUT", or "AUTO" (skip polling).
        Handles crash recovery via checkpoint.
        """
        task_id = task.get("task_id", "unknown")

        # Check for existing checkpoint (crash recovery)
        checkpoint = load_review_checkpoint(task_id, self._workspace)
        if checkpoint:
            review_id = checkpoint["review_id"]
            logger.info("Recovering review from checkpoint: task=%s review_id=%d",
                        task_id, review_id)
        else:
            # Submit for review
            result = await submit_for_review(
                task,
                callback_secret=self._review_callback_secret,
                api_key=self._review_api_key,
            )

            if result is None:
                # Submission failed — log and treat as auto-approved (fail-open)
                logger.warning("Review submission failed for %s, proceeding without review", task_id)
                return "APPROVED"

            routing = result.get("routing", "HUMAN")
            review_id = result.get("review_id")

            if routing == "AUTO":
                logger.info("Task %s auto-approved (routing=AUTO)", task_id)
                return "APPROVED"

            if not review_id:
                logger.error("No review_id in response for %s, proceeding", task_id)
                return "APPROVED"

            # Save checkpoint for crash recovery
            save_review_checkpoint(task_id, review_id, self._workspace)

        # Poll until review is resolved
        disposition = await poll_review_status(
            review_id,
            api_key=self._review_api_key,
        )

        # Clean up checkpoint
        clear_review_checkpoint(task_id, self._workspace)

        if disposition == "APPROVED":
            logger.info("Task %s approved by review, proceeding to execution", task_id)
        else:
            logger.warning("Task %s review result: %s", task_id, disposition)

        return disposition
    ```
